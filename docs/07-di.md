# DI

## Пример с PHP-DI

```php
use DI\ContainerBuilder;
use PhpSoftBox\Database\Configurator\DatabaseFactory;
use PhpSoftBox\Database\Connection\ConnectionManager;
use PhpSoftBox\Database\Connection\ConnectionManagerInterface;
use PhpSoftBox\Queue\Drivers\DatabaseDriver;
use PhpSoftBox\Queue\Drivers\DatabaseProgressStore;
use PhpSoftBox\Queue\DatabaseQueueProgressSchema;
use PhpSoftBox\Queue\DatabaseQueueSchema;
use PhpSoftBox\Queue\ProgressStoreInterface;
use PhpSoftBox\Queue\QueueInterface;
use PhpSoftBox\Queue\QueueJobHandlerInterface;
use PhpSoftBox\Queue\Worker;
use Psr\EventDispatcher\EventDispatcherInterface;

$builder = new ContainerBuilder();
$builder->addDefinitions([
    ConnectionManagerInterface::class => function (): ConnectionManagerInterface {
        $factory = new DatabaseFactory([
            'connections' => [
                'default' => 'main',
                'main' => [
                    'read' => ['dsn' => 'sqlite:///:memory:'],
                    'write' => ['dsn' => 'sqlite:///:memory:'],
                ],
            ],
        ]);

        return new ConnectionManager($factory);
    },
    QueueInterface::class => function (ConnectionManagerInterface $connections): QueueInterface {
        return new DatabaseDriver($connections, new DatabaseQueueSchema(), 'main');
    },
    ProgressStoreInterface::class => function (ConnectionManagerInterface $connections): ProgressStoreInterface {
        return new DatabaseProgressStore(
            connections: $connections,
            schema: new DatabaseQueueProgressSchema(),
            connectionName: 'main',
        );
    },
    EventDispatcherInterface::class => function (): EventDispatcherInterface {
        return new class () implements EventDispatcherInterface {
            public function dispatch(object $event): object
            {
                return $event;
            }
        };
    },
    Worker::class => function (
        QueueInterface $queue,
        ProgressStoreInterface $progressStore,
        EventDispatcherInterface $events,
    ): Worker {
        return new Worker(
            queue: $queue,
            maxAttempts: 3,
            progressStore: $progressStore,
            events: $events,
            progressStepPercent: 5,
        );
    },
    QueueJobHandlerInterface::class => function (): QueueJobHandlerInterface {
        return new class implements QueueJobHandlerInterface {
            public function handle(mixed $payload, \PhpSoftBox\Queue\QueueJob $job): void
            {
                // обработка
            }
        };
    },
]);
```
